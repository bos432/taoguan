2. 后台运营流程

flowchart TD A["管理员打开后台"] --> B["登录页"] B --> C{"账号密码正确?"} C -- "否" --> D["提示登录失败"] C -- "是" --> E["进入首页 / 工作台"] E --> F["商品管理"] E --> G["订单管理"] E --> H["会员管理"] E --> I["商家管理"] E --> J["系统管理"] E --> K["运营报表"] F --> Z["调用后台接口并写入数据库"] G --> Z H --> Z I --> Z J --> Z K --> Z Z --> R["结果回显"]

3. C 端用户流程

flowchart TD A["用户进入 H5 / 小程序 / APP"] --> B["首页 / 登录入口"] B --> C["协议默认未勾选"] C --> D{"是否勾选协议?"} D -- "否" --> E["拦截登录并提示先同意协议"] D -- "是" --> F["执行登录"] F --> G["浏览商品并下单"] G --> H["付款页展示商家收款码"] H --> I["上传付款凭证"] I --> J["后台审核"] J --> K["订单结果回显"]
